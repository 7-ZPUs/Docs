\section{Frontend - Navigazione}

\subsection{Descrizione}
Il modulo di navigazione realizza l'albero gerarchico del DIP ed e il punto di ingresso principale per la consultazione delle entita archivistiche. La gerarchia visualizzata segue il modello:
\begin{center}
DIP -> DocumentClass -> Process -> Document -> File
\end{center}

L'implementazione segue una separazione netta delle responsabilita:
\begin{itemize}
	\item \textbf{Facade Pattern}: \codeword{DipFacade} coordina stato, caricamenti e retry.
	\item \textbf{Gateway Pattern}: \codeword{IpcGateway} incapsula la comunicazione IPC con il backend.
	\item \textbf{Smart/Dumb Components}: componenti smart per orchestrazione, componenti dumb per sola presentazione del nodo.
	\item \textbf{State reattivo}: la UI reagisce ai cambiamenti di stato di root, cache, loading e errori locali.
\end{itemize}

\subsection{Flussi operativi principali}

\paragraph{Bootstrap albero root}
\begin{enumerate}[labelwidth=0.8cm,leftmargin=1cm]
	\item \codeword{AppShellComponent} o \codeword{DipExplorerComponent} avvia l'inizializzazione dei nodi root.
	\item \codeword{DipFacade.loadRootNodes(dipId)} imposta \codeword{phase=loading}.
	\item \codeword{IpcGateway.getRootDip} invoca \codeword{BROWSE\_GET\_DIP\_BY\_ID}.
	\item Il backend risolve la richiesta via \codeword{BrowsingIpcAdapter} e catena UseCase -> Outbound Port -> DAO.
	\item La facade aggiorna \codeword{phase=ready} e popola \codeword{rootNodes}.
\end{enumerate}

\paragraph{Espansione nodo}
\begin{enumerate}[labelwidth=0.8cm,leftmargin=1cm]
	\item \codeword{DipTree.toggleNode(node)} marca il nodo come espanso/collassato.
	\item Se il nodo non e in cache, \codeword{DipFacade.loadChildren(node)} avvia il caricamento.
	\item La facade aggiorna \codeword{loadingNodeIds} e pulisce eventuali errori pregressi sul nodo.
	\item \codeword{IpcGateway.getChildren} effettua dispatch per tipo nodo:
	\begin{itemize}
		\item dip -> \codeword{BROWSE\_GET\_DOCUMENT\_CLASS\_BY\_DIP\_ID}
		\item documentClass -> \codeword{BROWSE\_GET\_PROCESS\_BY\_DOCUMENT\_CLASS}
		\item process -> \codeword{BROWSE\_GET\_DOCUMENTS\_BY\_PROCESS}
		\item document -> \codeword{BROWSE\_GET\_FILE\_BY\_DOCUMENT}
		\item file -> nessun figlio
	\end{itemize}
	\item Al successo i figli sono salvati in \codeword{nodeCache}; al fallimento in \codeword{nodeChildrenErrors}.
\end{enumerate}

\paragraph{Selezione nodo e routing}
\begin{enumerate}[labelwidth=0.8cm,leftmargin=1cm]
	\item \codeword{DipTree.selectNode(node)} emette \codeword{nodeSelected}.
	\item AppShell/DipExplorer mappano il tipo nodo con \codeword{mapDipNodeTypeToDetailItemType()}.
	\item \codeword{buildDetailRoute(itemType, node.id)} costruisce la route target.
	\item Angular Router naviga verso \codeword{/detail/:itemType/:itemId}.
\end{enumerate}

\subsection{Contratti principali}

\paragraph{Stato della DipFacade}
\begin{itemize}
	\item \codeword{phase}: stato globale della feature (\codeword{ready|loading|idle}).
	\item \codeword{rootNodes}: nodi radice mostrati in albero.
	\item \codeword{nodeCache}: figli gia caricati per parent.
	\item \codeword{loadingNodeIds}: set dei nodi in caricamento.
	\item \codeword{nodeChildrenErrors}: errori locali per nodo.
	\item \codeword{rootError}: errore di bootstrap root.
\end{itemize}

\paragraph{Nodo di navigazione}
\begin{itemize}
	\item \codeword{name: string}
	\item \codeword{id: number}
	\item \codeword{type: dip|documentClass|process|document|file}
	\item \codeword{hasChildren: boolean}
	\item \codeword{timestamp?: string}
\end{itemize}

\paragraph{Firme metodi chiave}
\begin{itemize}
	\item \codeword{DipFacade.getState()}\\
	\codeword{: Signal<DipState>}
	\item \codeword{DipFacade.loadRootNodes(dipId: number)}\\
	\codeword{: Promise<void>}
	\item \codeword{DipFacade.loadChildren(target: number}\\
	\codeword{| DipTreeNode): Promise<void>}
	\item \codeword{IpcGateway.getRootDip(dipId: number)}\\
	\codeword{: Promise<DipTreeNode>}
	\item \codeword{IpcGateway.getChildren(parent: DipTreeNode)}\\
	\codeword{: Promise<DipTreeNode[]>}
	\item \codeword{DipTree.toggleNode(node: DipTreeNode)}\\
	\codeword{: void}
	\item \codeword{DipTree.retryNode(node: DipTreeNode)}\\
	\codeword{: void}
	\item \codeword{DipTree.selectNode(node: DipTreeNode)}\\
	\codeword{: void}
\end{itemize}

\subsection{Cache, naming e gestione errori}

\paragraph{Cache}
\begin{itemize}
	\item Cache nel renderer tramite \codeword{IpcCacheService}.
	\item TTL del gateway: \codeword{600000 ms}.
	\item Chiave nodo composta: \codeword{type:id}.
	\item Strutture correlate: \codeword{nodeCache}, \codeword{loadingNodeIds}, \codeword{nodeChildrenErrors}.
\end{itemize}

\paragraph{Naming}
\begin{itemize}
	\item Uso del nome preferito da DTO/metadata quando disponibile.
	\item Fallback per tipo nodo (DIP, Classe Documentale, Processo, Documento, File).
	\item Normalizzazione filename per i nodi file (ultimo segmento path).
\end{itemize}

\paragraph{Error handling}
\begin{itemize}
	\item Errori root: \codeword{rootError} e \codeword{phase=idle}.
	\item Errori figli: isolamento per nodo in \codeword{nodeChildrenErrors}.
	\item Retry nodo dedicato senza bloccare altri rami dell'albero.
	\item Normalizzazione errori IPC tramite \codeword{IpcErrorHandlerService}.
\end{itemize}
