\usecase{stampaSingoloDoc}{Stampa di un singolo documento}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Attore Secondario}: Stampante e Driver della stampante
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un documento (UC-\ref{selezionaDocumento})
        \item L'Utente sta visualizzando il documento
    \end{itemize}
    \item \textbf{Postcondizioni}: L'Utente ha stampato il documento
    \item \textbf{Flusso Principale}: L'Utente cliccando sull'apposita icona può selezionare la stampa del documento.
    \item \textbf{Flusso Alternativo}:
    \begin{itemize}
        \item L'Utente cerca di stampare un documento ma un errore esterno al programma e legato alla stampante interrompe l'azione.
        \item L'Utente cerca di stampare un Documento non stampabile (video, audio per esempio)
    \end{itemize}
\end{itemize}

\usecase{stampaInsiemeDoc}{Stampa di un insieme di documenti}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Attore Secondario}: Stampante e Driver della stampante
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente sta visualizzando un folder con più documenti o ha selezionato più documenti (UC-\ref{selezionaClasseDocumentale}, UC-\ref{selezionaProcesso}, UC-\ref{selezionaPiuDocumento})
    \end{itemize}
    \item \textbf{Postcondizioni}: L'Utente ha stampato tutti i documenti selezionati uno dopo l'altro
    \item \textbf{Flusso Principale}: L'Utente cliccando può selezionare tutti i documenti e quindi stamparli uno dopo l'altro.
    \item \textbf{Flusso Alternativo}:
    \begin{itemize}
        \item L'Utente cliccando può selezionare tutti i documenti ma un errore esterno al programma e legato alla stampante interrompe l'azione.
        \item L'Utente cerca di stampare uno o più documenti non stampabile (video, audio per esempio)
    \end{itemize}
\end{itemize}
