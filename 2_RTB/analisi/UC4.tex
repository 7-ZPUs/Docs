\usecase{stampaSingoloDoc}{Stampa di un singolo documento}
\begin{figure}[H]
      \centering
      \includegraphics[width=0.9\textwidth]{../assets/uml/UC22.png}
      \caption{\ref{stampaSingoloDoc} - Stampa di un singolo documento}
      \label{fig:uc_stampaSingoloDoc}
\end{figure}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Attore Secondario}: Stampante e Driver della stampante
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un documento
        \item L'Utente sta visualizzando il documento
    \end{itemize}
    \item \textbf{Postcondizioni}: L'Utente ha stampato il documento
    \item \textbf{Flusso Principale}:
        \begin{enumerate}
            \item L'Utente sceglie l'opzione di stampa del documento.
        \end{enumerate}
    \item \textbf{Flusso Alternativo}:
    \begin{itemize}
        \item L'Utente cerca di stampare un documento ma un errore esterno al programma e legato alla stampante interrompe l'azione (\ref{stampaFallita})
        \item L'Utente cerca di stampare un Documento non stampabile (video, audio per esempio) (\ref{stampaNonDisponibile})
    \end{itemize}
    \item \textbf{Estensioni}: 
    \begin{itemize}
        \item \ref{stampaFallita} Stampa fallita
        \item \ref{stampaNonDisponibile} Stampa Non Disponibile
    \end{itemize}
\end{itemize}

\usecase{stampaFallita}{Stampa fallita}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un elemento
        \item L'utente tenta di stampare uno o pi첫 documenti
    \end{itemize}
    \item \textbf{Postcondizioni}: Il sistema annulla l'operazione di stampa e mostra un messaggio di errore.
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item Il sistema annulla l'operazione di stampa.
        \item Il sistema mostra a video un messaggio di errore che indica il motivo del fallimento.
    \end{enumerate}
\end{itemize}
