\subsubsection{UC-3 - Salvataggio documento (LS)}

\subsubsection{UC 3.1 - Salvataggio di un singolo Documento}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente sta visualizzando un documento
    \item \textbf{Postcondizioni}: Il documento viene salvato in locale in una cartella selezionata dall'utente
    \item \textbf{Flusso Principale}: L'Utente cliccando può selezionare il salvataggio del documento corrente in una cartella del proprio file system, al di fuori del DIP.
    \item \textbf{Flusso Alternativo}: L'Utente cerca di salvare il documento nella cartella del DIP e questa operazione viene bloccata per garantire lo stato del DIP.
\end{itemize}

\subsubsection{UC 3.2 - Salvataggio di più Documenti}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente sta visualizzando più Documenti o una aggregazione di essi
    \item \textbf{Postcondizioni}: I documenti vengono salvati in locale in una cartella selezionata dall'utente
    \item \textbf{Flusso Principale}: L'Utente cliccando può selezionare più documenti di cui effettuare il salvataggio in una cartella del proprio file system
    \item \textbf{Flusso Alternativo}: L'Utente cerca di salvare i documenti nella cartella del DIP e questa operazione viene bloccata per garantire lo stato del DIP.
\end{itemize}
