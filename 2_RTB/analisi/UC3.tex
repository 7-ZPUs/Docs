\usecase{salvaDocumento}{Salvataggio di un documento}
\begin{figure}[H]
      \centering
      \includegraphics[width=0.6\textwidth]{../assets/uml/UC19.png}
      \caption{\ref{salvaDocumento} - Salvataggio di un documento}
      \label{fig:uc_salvaDocumento}
\end{figure}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un elemento
    \end{itemize}
    \item \textbf{Postcondizioni}: L'Utente ha salvato il documento in locale in una cartella selezionata
    \item \textbf{Flusso Principale}:
        \begin{enumerate}
            \item Il sistema apre un dialogo di selezione cartella
            \item L'utente seleziona la cartella di destinazione 
            \item Il sistema salva il File nella cartella selezionata
            \item Il sistema conferma il salvataggio con un messaggio "File salvato con successo in [percorso]".
        \end{enumerate}
    \item \textbf{Flusso Alternativo}:
    \begin{itemize}
            \item L'utente tenta di salvare il documento in una cartella: il sistema blocca l'operazione.
            \item L'utente annulla l'operazione: il sistema chiude il dialogo senza salvare.
        \end{itemize}
    \item \textbf{Estensioni}: \ref{salvataggioFallito} Salvataggio fallito
\end{itemize}

\usecase{salvataggioFallito}{Salvataggio fallito}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: 
    \begin{itemize}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un elemento
        \item L'utente ha scelto l'opzione di salvare uno o pi첫 documenti
    \end{itemize}
    \item \textbf{Postcondizioni}: Il sistema annulla l'operazione di salvataggio e mostra un messaggio di errore.
    \item \textbf{Flusso Principale}:
        \begin{enumerate}
            \item Il sistema annulla l'operazione di salvataggio.
            \item Il sistema mostra a video un messaggio di errore che indica il motivo del fallimento.
        \end{enumerate}
\end{itemize}
