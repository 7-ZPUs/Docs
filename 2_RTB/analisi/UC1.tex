\subsubsection{UC-1 - Accesso all'applicativo}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'utente ha scaricato e estratto il DIP
    \item \textbf{Postcondizioni}: L'utente ha acceduto all'applicativo e l'alberatura del DIP viene visualizzata
    \item \textbf{Flusso Principale}: L'utente avvia l'eseguibile \texttt{DIPReader.html}
    \item \textbf{Flusso Alternativo}: Nel caso l'utente selezioni un altro file all'interno del DIP, l'applicativo non viene avviato e viene aperto il file selezionato
\end{itemize}

\subsubsection{UC-1.1 - Visualizzazione tutte le Classi Documentali}

    \item \textbf{Precondizioni}: Accesso al DIP-Reader

\end{itemize}
