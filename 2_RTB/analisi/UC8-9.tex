
\subsubsection{UC-8 - Verifica integrità documento principale}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente ha selezionato un documento
    \end{enumerate}
    \item \textbf{Postcondizioni}: Visualizzazione di un report di integrità del documento
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item L'utente attiva la funzione di verifica di integrità del documento tramite apposito pulsante
        \item Viene effettuata la verifica di integrità del documento
        \item Il sistema restituisce una vista di report sull'integrità di conservazione del documento
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: La verifica fallisce e viene restituito un messaggio di errore
\end{itemize}

\subsubsection{UC-9 - Visualizzazione storia del DIP}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'utente ha avviato il programma e si trova nella radice del DIP
    \item \textbf{Postcondizioni}: L'utente visualizza la storia del DIP
    \item \textbf{Flusso Alternativo}: Viene restituito un messaggio di errore
\end{itemize}
