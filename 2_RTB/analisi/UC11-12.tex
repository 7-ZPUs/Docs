
\subsubsection{UC-11 - Consulta dettagli processo di conservazione AiP (D.V.)}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'utente ha avviato il programma
        \item L'utente ha selezionato un documento
        \item L'utente sta visualizzando le informazioni del relativo AiP
    \end{enumerate}
    \item \textbf{Postcondizioni}: L'utente consulta i dettagli operativi del processo di conservazione
    \item \textbf{Flusso Principale}: L'utente visualizza i dettagli operativi del processo di conservazione
    \item \textbf{Flussi Alternativi}:
    \begin{itemize}
        \item L'utente consulta anche i dettagli della sessione di versamento dell'AiP (UC13)
        \item L'utente consulta anche i dettagli della sessione di conservazione dell'AiP (UC14)
    \end{itemize}
    \item \textbf{Inclusioni}: UC12, UC13, UC14
\end{itemize}

\subsubsection{UC-12 - Consulta dettagli operativi (D.V.)}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'utente ha avviato il programma
        \item L'utente sta visualizzando i dettagli di un processo/sessione
    \end{enumerate}
    \item \textbf{Postcondizioni}:
    \begin{enumerate}
        \item L'utente visualizza la collocazione temporale del/la processo/sessione (inizio e fine)
        \item L'utente visualizza l'UUID dell'utente attivatore
        \item L'utente visualizza il canale tramite cui è stato/a avviato/a il processo/sessione
    \end{enumerate}
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item L'utente visualizza la collocazione temporale del/la processo/sessione (inizio e fine) (UC12.1)
        \item L'utente visualizza l'UUID dell'utente attivatore (UC12.2)
        \item L'utente visualizza il canale tramite cui è stato/a avviato/a il processo/sessione (UC12.3)
    \end{enumerate}
    \item \textbf{Inclusioni}: UC12.1, UC12.2, UC12.3
\end{itemize}
