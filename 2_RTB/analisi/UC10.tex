
\subsubsection{UC-10 - Tracciamento AiP di provenienza documento (D.V.)}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'utente ha avviato l'applicazione
        \item L'utente ha selezionato un documento
    \end{enumerate}
    \item \textbf{Postcondizioni}: L'utente ha ottenuto le informazioni sull'AiP di provenienza associate al documento: UUID e classe documentale.
    \item \textbf{Flusso Principale}: Il sistema mostra a video le informazioni sull'AiP:
    \begin{itemize}
        \item classe documentale AiP (UC10.1)
        \item UUID AiP (UC10.2)
    \end{itemize}
    \item \textbf{Flusso Alternativo}: L'utente visualizza anche i dettagli del processo di conservazione dell'AiP
    \item \textbf{Inclusioni}: UC10.1, UC10.2, UC11
\end{itemize}

\subsubsection{UC-10.1 - Visualizza classe documentale di appartenenza AiP (D.V.)}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'utente ha avviato l'applicazione
        \item L'utente ha selezionato un documento
        \item L'utente sta visualizzando i dettagli dell'AiP di provenienza del documento
    \end{enumerate}
    \item \textbf{Postcondizioni}: L'utente visualizza la classe documentale dell'AiP del documento selezionato
    \item \textbf{Flusso Principale}: Viene visualizzata la classe documentale dell'AiP del documento selezionato
\end{itemize}
