
\subsubsection{UC-5 - Verifica integrità DIP}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}:
    \begin{enumerate}
        \item L'Utente ha avviato l'applicazione
        \item L'Utente si trova nella cartella radice del DIP
    \end{enumerate}
    \item \textbf{Postcondizioni}: Visualizzazione di un report di integrità del DIP
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item L'utente attiva la funzione di verifica di integrità del DIP tramite apposito pulsante
        \item Viene effettuata la verifica di integrità delle classi documentali al suo interno
        \item Il sistema restituisce una vista di report sull'integrità di conservazione del DIP
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: La verifica fallisce per uno o più classi e viene restituito un messaggio di errore con la descrizione dei file non integri
\end{itemize}
