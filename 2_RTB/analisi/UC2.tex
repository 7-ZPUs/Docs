
\subsubsection{UC-2 - Ricerca Nel DIP}

\subsubsection{UC-2.1 - Ricerca Classe Documentale con filtri}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente sta visualizzando la lista delle Classi Documentali
    \item \textbf{Postcondizioni}: Elenco delle Classi Documentali corrispondenti ai filtri applicati
    \item \textbf{Flusso Principale}: L'Utente inserisce i filtri per la ricerca secondo i metadati delle Classi Documentali e visualizza i risultati della ricerca
    \item \textbf{Flusso Alternativo}: L'Utente visualizza un elenco vuoto/errore
\end{itemize}

\subsubsection{UC-2.2 - Ricerca Processo con filtri}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente sta visualizzando la lista dei Processi
    \item \textbf{Postcondizioni}: Elenco dei Processi corrispondenti ai filtri applicati
    \item \textbf{Flusso Principale}: L'Utente inserisce i filtri per la ricerca secondo i metadati di Processi e visualizza i risultati della ricerca
    \item \textbf{Flusso Alternativo}: L'Utente visualizza un elenco vuoto/errore
\end{itemize}

\subsubsection{UC-2.3 - Ricerca Documento con filtri}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'utente ha accesso alla funzionalità di ricerca
    \item \textbf{Postcondizioni}: Visualizzazione di un elenco di documenti che corrispondono ai criteri inseriti
    \item \textbf{Flusso Principale}: L'Utente imposta dei filtri e viene restituita la lista dei risultati
    \item \textbf{Flusso Alternativo}: Nessun risultato trovato o errore
\end{itemize}

\subsubsection{UC-2.3.1 - Ricerca per Campi Comuni}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente ha accesso alla funzionalità di ricerca
    \item \textbf{Postcondizioni}: Visualizzazione dei file all'interno della cartella contenenti il campo ricercato
    \item \textbf{Flusso Principale}: 
    \begin{enumerate}
        \item L'utente seleziona il campo di ricerca.
        \item L'utente compila il campo di ricerca.
        \item Viene restituita la lista dei documenti contenenti quel campo.
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: Il campo cercato non è valido o non è presente all'interno del DIP, si restituisce "La ricerca non ha prodotto alcun risultato"
\end{itemize}

\subsubsection{UC-2.3.1.1 - Ricerca per Campo Comune: Chiave Descrittiva}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente ha accesso alla funzionalità di ricerca e decide di cercare il documento in base alla Chiave Descrittiva, Oggetto e Parole Chiave.
    \item \textbf{Postcondizioni}: Visualizzazione dei file all'interno della cartella contenenti il campo ricercato
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item L'utente digita nella casella di ricerca i campi, con la precisazione che le Parole Chiave, non sono obbligatorie e quindi il documento che si cerca potrebbe non essere tra i risultati.
        \item Viene restituita la lista dei documenti nei quali la Chiave Descrittiva corrisponde al parametro ricercato.
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: Il campo cercato non è valido o non è presente all'interno del DIP, si restituisce "La ricerca non ha prodotto alcun risultato"
\end{itemize}

\subsubsection{UC-2.3.1.2 - Ricerca per Campo Comune: Soggetto}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente ha accesso alla funzionalità di ricerca e decide di cercare il documento in base ai Soggetti coinvolti.
    \item \textbf{Postcondizioni}: Visualizzazione dei file all'interno della cartella contenenti il campo ricercato
    \item \textbf{Flusso Principale}: L'utente sceglie i ruoli dei diversi soggetti tra:
    \begin{itemize}
        \item Amministrazione titolare con tipo soggetto: PAI (Amministrazioni Pubbliche Italiane)
        \item Amministrazioni partecipanti con tipi soggetto tra: PAI, PAE (Amministrazioni Pubbliche Estere)
        \item Assegnatario con tipo soggetto: AS
        \item Soggetto intestatario persona giuridica con tipi soggetto tra: PG (Organizzazioni), PAI, PAE
        \item Soggetto intestatario persona fisica con tipo soggetto: PF (Persona Fisica)
        \item RUP (Responsabile Unico del Procedimento) solo nel caso in cui la Tipologia Aggregazione sia Fascicolo
    \end{itemize}
    Viene restituita la lista dei documenti nei quali l'Id dell'Aggregazione corrisponde ai parametri ricercati.
    \item \textbf{Flusso Alternativo}: Il campo cercato non è valido o non è presente all'interno del DIP, si restituisce "La ricerca non ha prodotto alcun risultato"
\end{itemize}

\subsubsection{UC-2.3.1.2.1 - Ricerca per Campo Comune: Soggetti (Aggiunta nuovo soggetto)}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente ha accesso alla funzionalità di ricerca, decide di cercare il documento in base ai Soggetti e decide di aggiungere un nuovo soggetto come campo di ricerca
    \item \textbf{Postcondizioni}: Viene aggiunto un nuovo soggetto per l'aggregazione, per poter in seguito inserire i parametri per il nuovo soggetto
    \item \textbf{Flusso Principale}: 
    \begin{enumerate}
        \item L'utente può aggiungere un numero da 0 a n di soggetti per un'Aggregazione
        \item L'utente può inserire i parametri per il nuovo soggetto creato
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: Non selezionando l'aggiunta di un nuovo soggetto la ricerca può proseguire senza prendere in considerazione questo metadato
\end{itemize}

\subsubsection{UC-2.3.2.1.1 - Ricerca per Campo di un Documento Informatico e Amministrativo Informatico: Dati di Registrazione}
\begin{itemize}
    \item \textbf{Attore Primario}: Utente
    \item \textbf{Precondizioni}: L'Utente ha accesso alla funzionalità di ricerca, sceglie come tipo di documento il Documento Informatico o Amministrativo Informatico e decide di cercare il documento in base ai Dati di Registrazione, inteso come Tipologia di Flusso, Tipo di Registro, Data di Registrazione, Numero di Documento e Codice di Registro
    \item \textbf{Postcondizioni}: Visualizzazione dei file all'interno della cartella contenenti il campo ricercato
    \item \textbf{Flusso Principale}:
    \begin{enumerate}
        \item L'utente sceglie la Tipologia di Flusso tra: Uscita, Entrata, Interno.
        \item L'utente sceglie il Tipo di Registro tra: Nessuno, Protocollo Ordinario/Protocollo di Emergenza, Repertorio/Registro.
        \item L'utente inserisce la Data/Ora di Registrazione (nel caso di un documento protocollato tali parametri fanno riferimento alla protocollazione).
        \item L'utente inserisce il codice identificativo del Registro.
        \item Viene restituita la lista dei documenti nei quali i Dati di Registrazione corrispondono ai parametri ricercati.
    \end{enumerate}
    \item \textbf{Flusso Alternativo}: Il campo cercato non è valido o non è presente all'interno del DIP, si restituisce "La ricerca non ha prodotto alcun risultato"

    \item \textbf{Precondizioni}: L'utente ha accesso ad internet
    \item \textbf{Postcondizioni}: Visualizzazione di un elenco di documenti o insieme di documenti corrispondenti alla richiesta effettuata

    \item \textbf{Flusso Alternativo}:
    \begin{itemize}
        \item La ricerca non riporta alcun risultato, il documento o l'insieme di documenti richiesti non esistono all'interno del FAISS (e quindi del DIP)
        \item La connessione internet viene a mancare in fase di invio o ritorno della richiesta e quindi non è possibile portare a termine la richiesta
    \end{itemize}
\end{itemize}
