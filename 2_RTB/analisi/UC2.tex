\begin{itemize}
      \item \textbf{Attore Primario}: Utente
      \item \textbf{Precondizioni}: L'Utente sta visualizzando la lista delle Classi Documentali
      \item \textbf{Postcondizioni}: Elenco delle Classi Documentali corrispondenti ai filtri applicati
      \item \textbf{Flusso Principale}:
            \begin{enumerate}

      \item \textbf{Precondizioni}:
            \begin{enumerate}
                  \item L'Utente ha accesso alla funzionalità di ricerca
                  \item L'Utente sceglie come tipo di documento l'Aggregazione Documentale
                  \item L'Utente decide di aggiungere una nuova Assegnazione come campo per la ricerca.
            \end{enumerate}
      \item \textbf{Postcondizioni}: Viene aggiunta una nuova Assegnazione per l'aggregazione, per poter in seguito inserire i parametri per la nuova Assegnazione
      \item \textbf{Flusso Principale}:
            \begin{enumerate}
                  \item L'utente può aggiungere un numero da 0 a n Assegnazioni per un Aggregazione
                  \item L'utente può inserire i parametri per la nuova Aggregazione creata
                        (UC-2.3.2.2.8.1)
            \end{enumerate}
      \item \textbf{Flusso Alternativo}: Non selezionando l'aggiunta di una nuova Assegnazione la ricerca può proseguire senza prendere in considerazione questo metadato.
\end{itemize}

\subsubsection{UC-2.3.2.2.8.1 - Ricerca per Campo di un'Aggregazione Documentale: Assegnazione}
\begin{itemize}
      \item \textbf{Attore Primario}: Utente
      \item \textbf{Precondizioni}:
            \begin{enumerate}
                  \item L'Utente ha accesso alla funzionalità di ricerca
                  \item L'Utente sceglie come tipo di documento l'Aggregazione Documentale
                  \item L'Utente decide di aggiungere una nuova Assegnazione come campo per la ricerca.
            \end{enumerate}
      \item \textbf{Postcondizioni}: Viene aggiunta una nuova Assegnazione per l'aggregazione, per poter in seguito inserire i parametri per la nuova Assegnazione
      \item \textbf{Flusso Principale}:
            \begin{enumerate}
                  \item L'utente sceglie il Tipo Assegnazione tra: Per competenza, Per conoscenza.
                  \item L'utente inserisce il Soggetto Assegnatario come da UC-2.3.1.
                  \item L'utente inserisce la Data di Inizio dell'Assegnazione.
                  \item L'utente inserisce la Data di Fine dell'Assegnazione, se terminata.
                  \item Viene restituita la lista dei documenti nei quali i parametri di Verifica
                        corrispondono ai parametri ricercati (\nameref{risultatiRicerca}).
            \end{enumerate}
      \item \textbf{Flusso Alternativo}: Non selezionando l'aggiunta di una nuova Assegnazione la ricerca può proseguire senza prendere in considerazione questo metadato.
\end{itemize}

\subsubsection{UC-2.3.2.2.9 - Ricerca per Campo di un'Aggregazione Documentale: Progressivo}
\begin{itemize}
      \item \textbf{Attore Primario}: Utente
      \item \textbf{Precondizioni}:
            \begin{enumerate}
                  \item L'Utente ha accesso alla funzionalità di ricerca
                  \item L'Utente sceglie come tipo di documento l'Aggregazione Documentale
                  \item L'Utente decide di cercare l'aggregazione in base al campo Progressivo.
            \end{enumerate}
      \item \textbf{Postcondizioni}: Visualizzazione dei file all'interno della cartella DIP contenenti il campo ricercato
      \item \textbf{Flusso Principale}:
            \begin{enumerate}
                  \item L'utente inserisce il numero Progressivo dell'Aggregazione.
                  \item Viene restituita la lista dei documenti nei quali il Progressivo corrisponde al
                        parametro ricercato (\nameref{risultatiRicerca}).
            \end{enumerate}
      \item \textbf{Flusso Alternativo}: Il campo cercato non è presente all'interno del DIP. La ricerca non ha prodotto alcun risultato.(UC-)
\end{itemize}

\subsubsection{UC-2.4 - Visualizzazione Risultati di Ricerca} \label{risultatiRicerca}
\begin{itemize}
      \item \textbf{Attore Primario}: Utente
      \item \textbf{Precondizioni}: L'utente ha compilato la ricerca
      \item \textbf{Postcondizioni}: Visualizzazione di un elenco di documenti o insieme di documenti corrispondenti alla richiesta effettuata
      \item \textbf{Flusso Principale}: L'Utente descrive in poche parole nell'apposita searchbar che tipo di documento o insieme di documenti vuole ricercare all'interno del DIP
      \item \textbf{Flusso Alternativo}:
            \begin{itemize}
                  \item La ricerca non riporta alcun risultato, il documento o l'insieme di documenti
                        richiesti non esistono all'interno del FAISS (e quindi del DIP)
                  \item La connessione internet viene a mancare in fase di invio o ritorno della
                        richiesta e quindi non è possibile portare a termine la richiesta
            \end{itemize}
\end{itemize}
